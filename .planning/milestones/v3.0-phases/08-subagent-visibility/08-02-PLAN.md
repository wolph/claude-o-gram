**2. Modify `processEntry` to handle sidechain entries:**

Replace the current `if (entry.isSidechain) return;` with:
```typescript
if (entry.isSidechain) {
  // Subagent text output: emit via sidechain callback if registered
  if (!this.onSidechainMessage) return;
  if (entry.type !== 'assistant') return;
  if (entry.message.role !== 'assistant') return;

  // Extract text content (same logic as main chain)
  const content = entry.message.content;
  let text: string | null = null;
  if (typeof content === 'string') {
    text = content;
  } else if (Array.isArray(content)) {
    const textParts = (content as ContentBlock[])
      .filter((block): block is { type: 'text'; text: string } => block.type === 'text')
      .map((block) => block.text);
    text = textParts.length > 0 ? textParts.join('\n') : null;
  }

  if (text && text.trim().length > 0) {
    this.onSidechainMessage(text);
  }
  return;
}
```

This extracts text the same way as main-chain messages but routes it through the sidechain callback. The callback in index.ts will handle prefixing with the agent name.

**Important:** Do NOT change the existing main-chain `onAssistantMessage` logic. Only sidechain handling changes.
