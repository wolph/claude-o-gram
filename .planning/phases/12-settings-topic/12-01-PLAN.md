private save(): void {
  mkdirSync(dirname(this.filePath), { recursive: true });
  const json = JSON.stringify(data, null, 2);
  const tmpPath = this.filePath + '.tmp';
  writeFileSync(tmpPath, json, 'utf-8');
  renameSync(tmpPath, this.filePath);
}
private load(): void {
  if (!existsSync(this.filePath)) return;
  try {
    const raw = readFileSync(this.filePath, 'utf-8');
    const data = JSON.parse(raw);
    Object.assign(this.state, data);
  } catch { /* start with defaults */ }
}

Create `src/settings/bot-state-store.ts`:

1. Define and export `BotState` interface:
```typescript
export interface BotState {
  settingsTopicId: number;
  settingsMessageId: number;
  subagentOutput: boolean;
  defaultPermissionMode: string; // 'manual' | 'safe-only' | 'accept-all' | 'until-done'
}
```

2. Create `BotStateStore` class following the SessionStore atomic persistence pattern:
- Constructor takes `filePath: string` and `defaults: BotState`
- Stores `private state: BotState` initialized from defaults
- Calls `this.load()` in constructor to override defaults from disk
- `get(): BotState` -- returns current state (read-only snapshot)
- `update(partial: Partial<BotState>): void` -- merges partial into state, calls `this.save()`
- `private save(): void` -- atomic write: `mkdirSync` parent, `writeFileSync` to `.tmp`, `renameSync` to real path
- `private load(): void` -- if file exists, `readFileSync` + `JSON.parse`, merge into state with `Object.assign`. Catch and warn on corrupt files.

Import from `node:fs`: `writeFileSync, readFileSync, renameSync, existsSync, mkdirSync`
Import from `node:path`: `dirname`

Create `src/settings/runtime-settings.ts`:

1. Import `BotStateStore` and `BotState` from `./bot-state-store.js`

2. Create `RuntimeSettings` class:
- Constructor takes `private store: BotStateStore`
- Getter/setter pairs that delegate to BotStateStore:
  - `get subagentOutput(): boolean` -- returns `this.store.get().subagentOutput`
  - `set subagentOutput(value: boolean)` -- calls `this.store.update({ subagentOutput: value })`
  - `get defaultPermissionMode(): string` -- returns `this.store.get().defaultPermissionMode`
  - `set defaultPermissionMode(value: string)` -- calls `this.store.update({ defaultPermissionMode: value })`
  - `get settingsTopicId(): number` -- returns `this.store.get().settingsTopicId`
  - `set settingsTopicId(value: number)` -- calls `this.store.update({ settingsTopicId: value })`
  - `get settingsMessageId(): number` -- returns `this.store.get().settingsMessageId`
  - `set settingsMessageId(value: number)` -- calls `this.store.update({ settingsMessageId: value })`

Export both `RuntimeSettings` class.
